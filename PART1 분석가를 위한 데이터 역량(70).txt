18-1. select, insert, update, delete
select * from emp;
select distinct empname from emp;
select * from emp where empname='a';
select empname as e from emp;
select empname e from emp;

insert into emp values('a', 'b', 123);
update emp set empname='b';
update emp set empname='b' where empname='a';
delete * from emp;
delete * from emp where empname='a';

18-2. between, in, like, is null
select * from emp where empname between 40 and  50;
select * from emp where empname in(40, 50)
select * from emp where empname like 'a%'
select * from emp where empname is null;

18-3. order by
select * from emp order by empname asc;
